Exercício 1

While

  • Crie um programa que some todos os números digitados pelo usuário até que ele digite 0.

Solução

Exercício 1

  • Exercício 1: 
    #include <stdio.h>
int main() {
    int numero;
    int soma = 0;

    printf("Digite numeros para somar (digite 0 para encerrar):\n");
    scanf("%i", &numero);

    while (numero != 0) {
        soma += numero;
        scanf("%i", &numero);
    }

    printf("A soma dos numeros digitados é: %i\n", soma);

    return 0;
}

Exercício 2

do... while()

  • Conte quantos números pares e ímpares o usuário digitou até digitar 0.

Solução

Exercício 2

  • Exercício 2:
     #include <stdio.h>
int main() {
    int numero;
    int pares = 0;
    int impares = 0;

    do {
        printf("Digite um numero (0 para sair): ");
        scanf("%i", &numero);

        if (numero == 0) {
            break;
        }

        if (numero % 2 == 0) {
            pares++;
        } else {
            impares++;
        }

    } while (1);

    printf("\nQuantidade de numeros pares: %i\n", pares);
    printf("Quantidade de numeros impares: %i\n", impares);

    return 0;
}

Exercício 3

for()

  • Escreva um programa que calcule o fatorial de um número fornecido pelo usuário.

Solução

Exercício 3

  • Exercício 3:
    #include <stdio.h>
int main() {
    int n, i;
    int fatorial = 1;

    printf("Digite um numero inteiro positivo: ");
    scanf("%i", &n);

    if (n < 0) {
        printf("Nao existe fatorial de numero negativo.\n");
    } else {
        for (i = 1; i <= n; i++) {
            fatorial = fatorial * i;
        }
        printf("Fatorial de %i é: %d\n", n, fatorial);
    }

    return 0;
}
