Exercício 1, 2 e 3

While

  • 1. Escreva um programa que conte de 1 a 10 e mostre os números na tela.


    2. Crie um programa que some todos os números digitados pelo usuário até que ele digite 0.


    3. Crie um programa que some todos os números digitados pelo usuário até que ele digite a senha correta (escolha uma senha, ou use 1234)

Solução

Exercício 1, 2, 3

  • Exercício 1: 
    #include <stdio.h>

int main() {
    int i = 1;

    while (i <= 10) {
        printf("%i\n", i);
        i++;
    }

    return 0;
}
                                
    Exercício 2: 
    #include <stdio.h>

int main() {
    int numero;
    int soma = 0;

    printf("Digite numeros para somar (digite 0 para encerrar):\n");
    scanf("%i", &numero);

    while (numero != 0) {
        soma += numero;
        scanf("%i", &numero);
    }

    printf("A soma dos numeros digitados é: %i\n", soma);

    return 0;
}
                                
    Exercício 3: 
    #include <stdio.h>

int main() {
    int numero, soma = 0;
    int senha;

    printf("Digite numeros para somar.\n");
    printf("Quando quiser parar, digite a senha correta.\n");

    while (1) {
        printf("Digite um numero (ou a senha para encerrar): ");
        scanf("%i", &numero);

        if (numero == 1234) {
            break; // Encerra o laço se a senha for digitada
        }

        soma += numero;
    }

    printf("Soma total dos numeros digitados: %i\n", soma);

    return 0;
}

Exercício 1, 2 e 3

do... while()

  • 1. Faça um menu interativo e escolha uma opção para o usuário sair quando terminar de usar.


    2. Peça notas ao usuário (entre 0 e 10) até que ele digite -1 e calcule a média.


    3. Conte quantos números pares e ímpares o usuário digitou até digitar 0.

Solução

Exercício 1, 2, 3

  • Exercício 1:
     #include <stdio.h>

int main() {
    int opcao;

    do {
        printf("\n==== MENU ====\n");
        printf("1 - Dizer Ola\n");
        printf("2 - Mostrar um numero qualquer\n");
        printf("3 - Sair\n");
        printf("Escolha uma opcao: ");
        scanf("%d", &opcao);

        switch (opcao) {
            case 1:
                printf("Ola, usuario!\n");
                break;
            case 2:
                printf("Aqui está o numero: 42\n");
                break;
            case 3:
                printf("Saindo do programa...\n");
                break;
            default:
                printf("Opcao invalida. Tente novamente.\n");
        }
    } while (opcao != 3);

    return 0;
}
                                
    Exercício 2: 
    #include <stdio.h>

int main() {
    float nota, soma = 0;
    int contador = 0;

    do {
        printf("Digite uma nota entre 0 e 10 (ou -1 para sair): ");

    Exercício 3:
     #include <stdio.h>

int main() {
    int numero;
    int pares = 0;
    int impares = 0;

    do {
        printf("Digite um numero (0 para sair): ");
        scanf("%i", &numero);

        if (numero == 0) {
            break;

            pares++;
        } else {
            impares++;

Exercício 1, 2 e 3

for()

  • 1. Crie um programa que imprima os números ímpares de 1 a 20.


    2. Escreva um programa que imprima os números pares de 1 a 20.


    3. Escreva um programa que calcule o fatorial de um número fornecido pelo usuário.

Solução

Exercício 1, 2, 3

  • Exercício 1:
    #include <stdio.h>

int main() {
    int i;

    for (i = 1; i <= 20; i++) {
        if (i % 2 != 0) {
            printf("%i\n", i);
        }
    }

    return 0;
}
                                
    Exercício 2:
    #include <stdio.h>

int main() {
    int i;

    for (i = 1; i <= 20; i++) {
        if (i % 2 == 0) {
            printf("%i\n", i);
        }
    }

    return 0;
}
                                
    Exercício 3:
    #include <stdio.h>

int main() {
    int n, i;
    int fatorial = 1;

    printf("Digite um numero inteiro positivo: ");
    scanf("%i", &n);

    if (n < 0) {
        printf("Nao existe fatorial de numero negativo.\n");
    } else {
        for (i = 1; i <= n; i++) {
            fatorial = fatorial * i;
        }
        printf("Fatorial de %i é: %d\n", n, fatorial);
    }

    return 0;
}
